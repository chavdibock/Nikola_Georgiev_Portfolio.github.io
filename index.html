  const float* __restrict__ features,
  const float* __restrict__ train,
  float* dists, int n) {
  int tid = blockIdx.x * blockDim.x
           + threadIdx.x;

  dists[tid] = sqrtf(dist);

strategy.py 
def generate_signal(self, df):
  macd, sig = ta.MACD(df["close"])
  vwap = (
    (df["close"] * df["volume"]).cumsum()
    / df["volume"].cumsum()
  )
  # MACD crossover + price above VWAP
  go_long = (macd > sig) & \
            (df["close"] > vwap)
  return Signal.BUY if go_long.iloc[-1] \
         else Signal.HOLD
